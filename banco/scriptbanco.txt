create database Customizacao;
/* drop database Customizacao;*/
use Customizacao;

create table banco(
id int primary key auto_increment,
nome varchar(50));

create table cargo(
id int primary key auto_increment,
nome varchar(30),
codigo int);

create table Agencia(
id int primary key auto_increment,
numeroAgencia int,
fkBanco int,
foreign key (fkBanco) references Banco(id));

create table usuario (
id int primary key auto_increment,
email varchar(50),
senha varchar(255),
fkCargo int,
foreign key (fkCargo) references cargo(id),
fkAgencia int,
foreign key (fkAgencia) references Agencia(id));

create table Caixa (
id int primary key auto_increment,
fkAgencia int);

create table dadoEnergia (
id int primary key auto_increment,
fkCaixa int,
consumo decimal(4,2), 
plano varchar(30),
momento DATETIME,
foreign key (fkCaixa) references Caixa(id));
